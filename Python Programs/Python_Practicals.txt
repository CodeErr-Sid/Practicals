rev=0
while(num>0):
    digit = num%10
    rev = (rev*10) + digit
    num = num//10
print(f"Reverse Of The Number: {rev}")

Output:
Enter Any +ve Integer: 123
Reverse Of The Number: 321

# 3. Python Scripts

num = int(input("Enter Any Number: "))
rev=0
temp=num
while(num>0):
    digit = num%10
    rev = (rev*10)+digit
    num = num//10
if(rev==temp):
    print("Palindrome")
else:
    print("Not a Palindrome")

Output:
Enter Any Number: 343
Palindrome

b = int(input("Enter 2nd Number: "))
c = int(input("Enter 3rd Number: "))
def Find_max(a,b,c):
    if a>b and a>c:
        return a
    elif b>a and b>c:
        return b
    else:
        return c
max_num = Find_max(a,b,c)
print(f"Maximum Of The Three Numbers is: {max_num}")

Output:

Enter 2nd Number: 4
Enter 3rd Number: 5

# 7.Recursion & Parameter Passing

def fibo(n):
    if n<=1:
        return n
    else:
        return(fibo(n-1)+fibo(n-2))
