12. Branching And Looping

Source Code:

using System;
					
public class Program
{
	public static void Main(String[] args)
	{
        for(int i=0; i<10; i++)
        {
         Console.WriteLine("i={0}",i);
          if(i>=5){
           goto loopst;
          }
          else{
           continue;
          loopst: {break;}
          }
        }
    Console.ReadLine();
	}
}
